Acoustic Tracing Interpolation to Grid

using OceanSonar
using Plots
using .Threads

@time "all" begin
    begin
        models = list_models(Scenario)
        for m in eachindex(models)
            model = models[m]
            scen = Scenario(model)
            prop = @time "$model" Propagation(:trace, scen,
                angles = critical_angles(scen)
            )
            fig = visual(prop)
            savefig(fig, "prop_" * modelsnake(scen.model) * ".svg")
        end
    end
end
